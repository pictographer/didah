614 void readEdit(char* buffer, bool& isNotTooLate,
615  const char* end, IntPredicate isIncluded, uint32_t timeout)
616 {
617  const char ASCII_BS = '\b';
618  const char ASCII_DEL = '\x7f';
619  char* p(buffer);
620  elapsedMicros elapsed;
621  do {
622  if (Serial.available()) {
623  char in(Serial.read());
624  if (isIncluded(in)) {
625  *p = in;
626  Serial.write(*p);
627  ++p;
628  } else if (in == '\r') {
629  *p = 0;
630  break;
631  } else if (p != buffer && (in == ASCII_BS || in == ASCII_DEL)) {
632  Serial.write(ASCII_BS); // back up one
633  Serial.write(' '); // erase by writing a space
634  Serial.write(ASCII_BS); // back over the space
635  *p = 0;
636  --p;
637  }
638  }
639  isNotTooLate = elapsed < timeout;
640  } while (p != end && isNotTooLate);
641  if (isNotTooLate || p != buffer) {
642  *p = 0;
643  }
644 }
