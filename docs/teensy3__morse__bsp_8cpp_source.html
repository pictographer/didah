210 /// Turns out by the time user code runs, it's too late to change any
211 /// of these clock registers.
212 ///
213 /// From
214 /// http://forum.pjrc.com/threads/724-32-768-KHz-crystal-and-battery-for-Teensy-3-0-RTC-questions?highlight=RTC_TPR
215 /// The bit to use is RTC_SR_TIF, but you can't easily do this because
216 /// the startup code checks that bit and tries to initialize the
217 /// RTC. By the time your program is running, it's too late to
218 /// discover what that bit was before the startup code ran.
219 ///
220 /// There is a startup_early_hook() function you could implement to
221 /// run your own code very early in the startup. Pretty much nothing
222 /// is initialized at that point. The most viable solution would be to
223 /// grab the value of RTC_SR and put it into a static variable. But
224 /// this code runs before variables are initialized, so you'll have to
225 /// use an attribute to put that variable into the .noinit section to
226 /// prevent the later startup code from overwriting it. Then you could
227 /// check that variable's RTC_SR_TIF bit once your program is
228 /// running. Or if you don't need to do much, you could perhaps do
229 /// everything you need in that extremely early startup code.... but
230 /// that's quite difficult because so much stuff ordinary programming
231 /// depends upon hasn't been initialized yet.
232 void describeRTC() {
233  printLabelValueUnits("RTC_TSR ", RTC_TSR, " time in seconds");
234  printLabelValueUnits("RTC_CR & RTC_CR_CLKO ",
235  !!(RTC_CR & RTC_CR_CLKO), " 1=gated");
236  printLabelValueUnits("RTC_CR & RTC_CR_OSCE ",
237  !!(RTC_CR & RTC_CR_OSCE), " 1=enabled");
238  printLabelValueUnits("RTC_SR & RTC_SR_TCE ",
239  !!(RTC_SR & RTC_SR_TCE), " 1=enabled");
240  printLabelValueUnits("RTC_SR & RTC_SR_TIF ",
241  !!(RTC_SR & RTC_SR_TIF), " 1=invalid");
242  printLabelValueUnits("RTC_SR & RTC_SR_TCE ",
243  !!(RTC_SR & RTC_SR_TCE), " 1=prescale enabled");
244  printLabelValueUnits("RTC_TPR ", RTC_TPR, " prescaler");
245 }

247 /// Alas, all the flags I can find give no indication whether the RTC
248 /// external clock crystal is or is not present. The RTC prescaler
249 /// should increment approximately once every 31 microseconds, so we
250 /// sample the prescaler, wait 33 microseconds, and sample again. If
251 /// the value is unchanged, there is no RTC external crystal, but if
252 /// the prescaler value has changed, then there is.
253 ///
254 /// The flags I looked at were the following:
255 /// !!(RTC_CR & RTC_CR_CLKO)
256 /// !!(RTC_CR & RTC_CR_OSCE)
257 /// !!(RTC_SR & RTC_SR_TCE)
258 /// !!(RTC_SR & RTC_SR_TIF)
259 /// !!(RTC_SR & RTC_SR_TCE)
260 bool hasRTC() {
261  uint32_t prescaler(RTC_TPR);
262  delayMicroseconds(33);
263  return prescaler != RTC_TPR;
264 }
