21 /// passed MorseToken instances so that a single action can handle

62 ///\brief Score the given pattern against the state of the symbol stack.
63 /**
64 The maximum score is 255 and decreases with increased distance
65 from the pattern.
66 
67 \li \c A-Z, \c 0-9, punctuation - Match a literal character
68 \li \c ^CQ^ - Match a prosign
69 \li space - Match a word-length or greater pause
70 \li \c a - Match a single letter from A-Z
71 \li \c c - Match any character
72 \li \c d - Match a single digit
73 \li \c n - Match a contiguous sequence of digits
74 
75 With the exception of \c n, all the pattern elements match one
76 Morse token. The following punctuation is accepted:
77 \verbatim
78 !"$&'()+,-./:;=?@_ \endverbatim
79 
80 The pattern is compared to the stack starting at the end of the
81 pattern and the top of the stack. There is no support for
82 backtracking. All pattern elements are greedy.
83 */
84 uint8_t matchScore(const char* pattern);
