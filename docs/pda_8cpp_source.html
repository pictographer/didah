475 /// Generate a random digit with the probability distribution implied
476 /// by the digitErr histogram.
477 //
478 /// The sum of the entries in digitErr always equals digitErrCount. A
479 /// random ordinal number in the range (0..digitErrCount) is compared
480 /// to the cumulative distribution (calculated dynamically from the
481 /// ordinary histogram. As long as the random number is less than the
482 /// cumulative distribution, the search continues.
483 uint8_t getDigitByWeight() {
484  uint8_t a(random(digitErrCount));
485  uint8_t result(9);
486  for (uint8_t i(0), sum(0); i < 10; ++i) {
487  sum += digitErr[i] + 1;
488  if (a < sum) {
489  result = i;
490  break;
491  }
492  }
493  return result;
494 }

517 /// The top three entries on the stack are two digits to add and their
518 /// proported sum mod 10. Transmit Y/N to indicate whether or not the
519 /// sum is correct. If the sum is not correct, transmit the correct
520 /// sum. Update the weights to bias future random number selection.
521 void gradeSum(MorseToken) {
522  if (2 < symbolStackSize()) {
523  uint8_t s1(s(1).m2i());
524  uint8_t s2(s(2).m2i());
525  uint8_t s0(s(0).toChar());
526  uint8_t sum((s1 + s2) % 10);
527  Serial.print(s2);
528  Serial.print(" ");
529  Serial.print(s1);
530  Serial.print(" ");
531  Serial.print(s0);
532  Serial.print(" sum: ");
533  Serial.println(sum);
534  bool correct(s(0).m2i() == sum);
535  if (correct) {
536  txString("Y");
537  // Decrease the error counts associated with the digits the user
538  // just added incorrectly, taking care not to underflow.
539  uint8_t d1(!!digitErr[s1]);
540  uint8_t d2(!!digitErr[s2]);
541  digitErr[s1] -= d1;
542  digitErr[s2] -= d2;
543  digitErrCount -= (d1 + d2);
544  Serial.print("Decrementing ");
545  Serial.print(s1);
546  Serial.print(". New weight is: ");
547  Serial.println(digitErr[s1]);
548  Serial.print("Decrementing ");
549  Serial.print(s2);
550  Serial.print(". New weight is: ");
551  Serial.println(digitErr[s2]);
552  } else {
553  // 012345678901
554  char buffer[] = "N a + b = c";
555  buffer[3] = s2 + '0';
556  buffer[7] = s1 + '0';
557  buffer[11] = sum + '0';
558  txString(buffer);
559  // Increase the error counts associated with the digits the user
560  // just added incorrectly, taking care not to overflow.
561  uint8_t d1(3 * (digitErr[s1] - 3 < 255));
562  uint8_t d2(3 * (digitErr[s2] - 3 < 255));
563  digitErr[s1] += d1;
564  digitErr[s2] += d2;
565  digitErrCount += d1 + d2;
566  Serial.print("Incrementing ");
567  Serial.print(s1);
568  Serial.print(". New weight is: ");
569  Serial.println(digitErr[s1]);
570  Serial.print("Incrementing ");
571  Serial.print(s2);
572  Serial.print(". New weight is: ");
573  Serial.println(digitErr[s2]);
574  }
575  // If the user plays long enough without a reset to saturate two
576  // of the digits chosen at random, divide all the error counts
577  // by 2.
578  if (digitErr[s1] == 255 && digitErr[s2] == 255) {
579  for (uint8_t i(0); i < 10; ++i) {
580  digitErr[i] <<= 1;
581  }
582  digitErrCount <<= 1;
583  Serial.println("Halving all weights.");
584  }
585  Serial.print("Digit error count is: ");
586  Serial.println(digitErrCount);
587  popN(3);
588  } else {
589  txError();
590  }
591 }
