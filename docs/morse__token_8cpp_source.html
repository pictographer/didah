7 /// Morse code for letters encoded as an implicit binary tree.
8 /// Letters are written in breadth first order. Left child
9 /// corresponds to dit; right child corresponds to dah.
10 /// Caret is used for undefined (or incomplete) codes.
11 /// The root of the tree acts as a sentinel to simplify the process of
12 /// converting the loop index into our left-justified code scheme.
13 ///
14 /// Here are some examples to show how it works:
15 ///
16 /// Index | Letter | Morse Code | Notes
17 /// ----------|--------|--------------|------
18 /// \c 000000 | N/A | N/A | Invalid code
19 /// \c 000001 | space | N/A | Sentinel followed by zero bits (represents a pause)
20 /// \c 000010 | \c E | `.` | Sentinel followed by a dit
21 /// \c 000011 | \c T | `_` | Sentinel followed by a dah
22 /// \c 000100 | \c I | `. .` | Sentinel followed by two dits
23 /// \c 000101 | \c A | `. _` | And so on...
24 /// \c 000110 | \c N | `_ .` | &nbsp;
25 /// \c 000111 | \c M | `_ _` | &nbsp;
26 /// \c 001000 | \c S | `. . .` | &nbsp;
27 /// \c 001001 | \c U | `. . _` | &nbsp;
28 /// \c 001010 | \c R | `. _ .` | &nbsp;
29 /// \c 001011 | \c W | `. _ _` | &nbsp;
30 /// \c 001100 | \c D | `_ . .` | &nbsp;
31 /// \c 001101 | \c K | `_ . _` | &nbsp;
32 /// \c 001110 | \c G | `_ _ .` | &nbsp;
33 /// \c 001111 | \c O | `_ _ _` | &nbsp;

66  // sign is recognized as a six symbol prefix of its seven symbol

85 /// Convert a single ASCII character into a Morse code token.
86 MorseToken a2m(char a) {
87  uint8_t letterCode(0);
88  ptrdiff_t key(strchr(morseTree, a) - morseTree);
89  if (0 < key && morseTree[key] != MORSE_UNDEFINED_CHAR) {
90  // Add the sentinel to the right.
91  letterCode = key << 1 | 1;
92  while (!(letterCode & MORSE_MSB)) {
93  letterCode <<= 1;
94  }
95  letterCode <<= 1;
96  } else {
97  // The codes below would require more rows in morseTree and
98  // the rows would be mostly empty, so they are handled here.
99  switch (a) {
100  case '\n': letterCode = MORSE_AA; break;
101  case '&': letterCode = MORSE_AMPER; break;
102  case ')': letterCode = MORSE_CLOSEP; break;
103  case '$': letterCode = MORSE_DOLLAR; break;
104  default:
105  ; // Nothing more to do since letterCode is initialized to 0
106  // (invalid).
107  }
108  }
109  return MorseToken(letterCode);
110 }

123 /**
124 Given an ASCII string representing a prosign, return a
125 code composed by concatenating the morse code for each
126 character in the prosign without inter-symbol gaps.
127 
128 For example "AB" yields
129 
130 String | Morse Code | Binary
131 -------|---------------------------|-------
132 \c A | \c . _ | \c 01100000
133 \c B | \c &nbsp;&nbsp;_ . . . | \c &nbsp;&nbsp;10001000
134 \c AB | \c . _ _ . . . | \c 01100010
135 
136 If the translation is longer than will fit in MorseToken, the return
137 value will be the longest suffix of the input that does fit. All the
138 prosigns I'm aware of fit without truncation if MorseToken is 16 bits
139 or more. Most fit in 7 bits with the notable exception of
140 \c MORSE_ERROR, which requires 9 bits (8 symbols + 1 sentinel).
141 Unfortunately, there are many ways of denoting 8 dits, so the special
142 case isn't so easy to code, e.g. \c ^HH^, \c ^5S^, \c ^S5^, \c ^IIII^,
143 \c ^IIH^, \c ^EEEEEEEE^, etc.
144 */
145 MorseToken a2p(const char* prosign) {
146  uint8_t letterCode(MorseToken(prosign[0], MorseToken::Char));
147  uint8_t t(letterCode);
148  size_t length(0);
149  while (t) { ++length; t <<= 1; }
150  const char* p(prosign + 1);
151  while (*p) {
152  char a(*p);
153  uint8_t m(MorseToken(a, MorseToken::Char));
154  // Erase the sentinel (always 1).
155  letterCode ^= 1 << (wordBits - length);
156  // Shift the new symbol and or it in after the previous.
157  letterCode |= m >> (length - 1);
158  // Update the length.
159  while (m) { ++length; m <<= 1; }
160  ++p;
161  }
162  return MorseToken(letterCode);
163 }
